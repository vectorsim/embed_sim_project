1. Introduction

This guide demonstrates how to use CodeGenStart and CodeGenEnd in the EmbedSim framework to automatically generate C code from a Python-based three-phase signal processing chain.

What You'll Learn:
  • How to mark regions for code generation
  • How to generate C headers and wrappers
  • How to implement the C function
  • How to switch between Python and C backends
  • How to deploy to embedded targets (AURIX, Cortex-M4)

3. Step 1: Pure Python Version

First, let's create a pure Python version to understand what we're replacing.

3.1 Create the Python Script

File: code/01_python_version.py

#!/usr/bin/env python3
"""
Pure Python Version - No Code Generation
"""

import sys
import numpy as np
import matplotlib.pyplot as plt
from pathlib import Path

# Add parent directory to path
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

from embedsim.source_blocks import ThreePhaseGenerator
from embedsim.processing_blocks import VectorGain
from embedsim.dynamic_blocks import VectorEnd
from embedsim.simulation_engine import EmbedSim, ODESolver

def run_python_version():
    """Run pure Python simulation"""
    print("\n" + "="*60)
    print("PURE PYTHON VERSION")
    print("="*60)
    
    # Create blocks
    generator = ThreePhaseGenerator("source", amplitude=10.0, freq=50.0)
    gain = VectorGain("gain", gain=2.0)
    sink = VectorEnd("sink")
    
    # Connect
    generator >> gain
    gain >> sink
    
    # Create simulation
    sim = EmbedSim(sinks=[sink], T=0.04, dt=0.0001, solver=ODESolver.RK4)
    
    # Add to scope
    sim.scope.add(generator, label="original")
    sim.scope.add(gain, label="amplified")
    
    # Run
    print("\nRunning simulation...")
    sim.run(verbose=True)
    
    # Plot
    plt.figure(figsize=(12, 5))
    t = np.array(sim.scope.t)
    orig = sim.scope.get_signal("original", 0)
    amp = sim.scope.get_signal("amplified", 0)
    
    plt.plot(t, orig, 'b-', label="Original (10V)", linewidth=1.5)
    plt.plot(t, amp, 'r--', label="Amplified (20V)", linewidth=1.5)
    plt.xlabel("Time [s]")
    plt.ylabel("Amplitude [V]")
    plt.title("Three-Phase Signal - Phase U Only")
    plt.legend()
    plt.grid(True)
    plt.xlim(0, 0.04)
    plt.savefig("python_version.png")
    plt.show()
    
    print("\n✓ Python version completed successfully")
    return sim

if __name__ == "__main__":
    run_python_version()

4. Step 2: Add Code Generation Markers

Now we'll modify the example to include CodeGenStart and CodeGenEnd markers.

4.1 Create the Marked Version

File: code/02_with_markers.py

#!/usr/bin/env python3
"""
Version with Code Generation Markers
"""

import sys
import numpy as np
from pathlib import Path

sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

from embedsim.source_blocks import ThreePhaseGenerator
from embedsim.processing_blocks import VectorGain
from embedsim.dynamic_blocks import VectorEnd
from embedsim.simulation_engine import EmbedSim, ODESolver
from embedsim.code_generator import CodeGenStart, CodeGenEnd

def create_marked_version():
    """Create version with code generation markers"""
    print("\n" + "="*60)
    print("VERSION WITH CODE GENERATION MARKERS")
    print("="*60)
    
    # Create blocks
    generator = ThreePhaseGenerator("source", amplitude=10.0, freq=50.0)
    
    # MARK THE REGION FOR CODE GENERATION
    cg_start = CodeGenStart("region_start")  # Input boundary
    gain = VectorGain("gain", gain=2.0)       # Block to be replaced
    cg_end = CodeGenEnd("region_end")        # Output boundary
    
    sink = VectorEnd("sink")
    
    # Connect with markers
    generator >> cg_start    # Signal enters region
    cg_start >> gain         # Processing inside region
    gain >> cg_end           # Signal leaves region
    cg_end >> sink           # Continue to sink
    
    print("\nBlock Diagram:")
    print("  [source] → [cg_start] → [gain] → [cg_end] → [sink]")
    print("              <─── CODE GENERATION REGION ───>")
    
    # Run one step to determine signal sizes
    sim = EmbedSim(sinks=[sink], T=0.001, dt=0.0001, solver=ODESolver.RK4)
    sim.run(verbose=False)
    
    print("\nInput signals to region:")
    cg_start.print_signal_info()
    
    print("\nOutput signals from region:")
    cg_end.print_signal_info()
    
    return generator, cg_start, gain, cg_end, sink, sim

if __name__ == "__main__":
    create_marked_version()

4.2 Understanding the Markers

Block Purpose
CodeGenStart Marks input boundary. All upstream blocks become fields in the input struct.
CodeGenEnd Marks output boundary. All upstream blocks become fields in the output struct.

5. Step 3: Generate C Code

5.1 Create Code Generation Script

File: code/03_generate_code.py

#!/usr/bin/env python3
"""
Generate C code from marked region
"""

import sys
from pathlib import Path

sys.path.insert(0, str(Path(__file__).resolve().parent.parent))

from embedsim.source_blocks import ThreePhaseGenerator
from embedsim.processing_blocks import VectorGain
from embedsim.dynamic_blocks import VectorEnd
from embedsim.simulation_engine import EmbedSim, ODESolver
from embedsim.code_generator import CodeGenStart, CodeGenEnd

def generate_code():
    """Generate C code from marked region"""
    print("\n" + "="*60)
    print("GENERATING C CODE")
    print("="*60)
    
    # Create blocks with markers
    generator = ThreePhaseGenerator("source", amplitude=10.0, freq=50.0)
    cg_start = CodeGenStart("region_start")
    gain = VectorGain("gain", gain=2.0)
    cg_end = CodeGenEnd("region_end")
    sink = VectorEnd("sink")
    
    # Connect
    generator >> cg_start
    cg_start >> gain
    gain >> cg_end
    cg_end >> sink
    
    # Run one step to determine signal sizes
    print("\nRunning one step to determine signal sizes...")
    sim = EmbedSim(sinks=[sink], T=0.001, dt=0.0001, solver=ODESolver.RK4)
    sim.run(verbose=False)
    
    # Generate code
    print("\nGenerating C code stubs...")
    files = cg_end.generate_pyx_stub(
        cg_start=cg_start,
        block_name="three_phase_processor",
        output_dir="../generated_code",
        write_files=True
    )
    
    print("\nGenerated files:")
    for name, content in files.items():
        print(f"  - {name}: {len(content)} bytes")
    
    print("\n✓ Code generation complete!")
    return files

if __name__ == "__main__":
    generate_code()

 * three_phase_processor.c
 * 
 * C implementation of three-phase signal processing.
 * 
 * Input:  source[3] - three-phase signals [ia, ib, ic]
 * Output: gain[3]   - amplified signals
 */

#include "three_phase_processor.h"
#include 

void three_phase_processor_compute(const InputSignals* in, OutputSignals* out) {
    /**
     * Simple gain of 2.0 on all three phases
     * 
     * Mathematical operation:
     *   out->gain[i] = in->source[i] * 2.0
     */
    
    // Method 1: Manual loop
    for (int i = 0; i < 3; i++) {
        out->gain[i] = in->source[i] * 2.0;
    }
    
    /* 
     * Method 2: Unrolled (faster for small arrays)
     * out->gain[0] = in->source[0] * 2.0;
     * out->gain[1] = in->source[1] * 2.0;
     * out->gain[2] = in->source[2] * 2.0;
     */
}

8. Step 6: Test C Version

8.1 Create Test Script

File: code/04_test_c_version.py

#!/usr/bin/env python3
"""
Test the compiled C version
"""

import sys
import numpy as np
import matplotlib.pyplot as plt
from pathlib import Path

# Add paths
sys.path.insert(0, str(Path(__file__).resolve().parent.parent))
sys.path.insert(0, str(Path(__file__).resolve().parent.parent / "generated_code"))

from embedsim.source_blocks import ThreePhaseGenerator
from embedsim.dynamic_blocks import VectorEnd
from embedsim.simulation_engine import EmbedSim, ODESolver
from three_phase_processor_simblock import ThreePhaseProcessorSimBlock

def test_c_version():
    print("\n" + "="*70)
    print("TESTING C VERSION")
    print("="*70)
    
    # Create source
    source = ThreePhaseGenerator("source", amplitude=10.0, freq=50.0)
    
    # Create both Python and C versions
    processor_c = ThreePhaseProcessorSimBlock("processor_c", use_c_backend=True)
    processor_py = ThreePhaseProcessorSimBlock("processor_py", use_c_backend=False)
    
    # Sinks
    sink_c = VectorEnd("sink_c")
    sink_py = VectorEnd("sink_py")
    
    # Connect
    source >> processor_c >> sink_c
    source >> processor_py >> sink_py
    
    # Run simulations
    print("\nRunning C version...")
    sim_c = EmbedSim(sinks=[sink_c], T=0.04, dt=0.0001, solver=ODESolver.RK4)
    sim_c.scope.add(processor_c, label="c_output")
    sim_c.run(verbose=False)
    
    print("Running Python version...")
    sim_py = EmbedSim(sinks=[sink_py], T=0.04, dt=0.0001, solver=ODESolver.RK4)
    sim_py.scope.add(processor_py, label="py_output")
    sim_py.run(verbose=False)
    
    # Get data
    t_c = np.array(sim_c.scope.t)
    t_py = np.array(sim_py.scope.t)
    c_data = sim_c.scope.get_signal("c_output", 0)
    py_data = sim_py.scope.get_signal("py_output", 0)
    
    # Plot
    plt.figure(figsize=(12, 8))
    
    plt.subplot(2, 1, 1)
    plt.plot(t_c, c_data, 'r-', label="C Version", linewidth=2)
    plt.plot(t_py, py_data, 'b--', label="Python Version", linewidth=2, alpha=0.7)
    plt.ylabel("Amplitude")
    plt.title("C vs Python Implementation")
    plt.legend()
    plt.grid(True)
    plt.xlim(0, 0.04)
    
    plt.subplot(2, 1, 2)
    diff = c_data - py_data
    plt.plot(t_c, diff, 'g-', label="Difference")
    plt.xlabel("Time [s]")
    plt.ylabel("Difference")
    plt.title(f"Max Difference: {np.max(np.abs(diff)):.2e}")
    plt.legend()
    plt.grid(True)
    plt.xlim(0, 0.04)
    
    plt.tight_layout()
    plt.savefig("c_vs_python.png")
    plt.show()
    
    print(f"\nMax difference: {np.max(np.abs(diff)):.2e}")
    if np.max(np.abs(diff)) < 1e-10:
        print("✓ C implementation matches Python exactly!")
    else:
        print("⚠ C implementation differs from Python")
    
    return c_data, py_data

if __name__ == "__main__":
    test_c_version()

9. Step 7: Performance Comparison

9.1 Create Performance Test

File: code/05_performance_test.py

#!/usr/bin/env python3
"""
Performance comparison between Python and C versions
"""

import sys
import time
import numpy as np
import matplotlib.pyplot as plt
from pathlib import Path

sys.path.insert(0, str(Path(__file__).resolve().parent.parent))
sys.path.insert(0, str(Path(__file__).resolve().parent.parent / "generated_code"))

from embedsim.source_blocks import ThreePhaseGenerator
from embedsim.processing_blocks import VectorGain
from three_phase_processor_simblock import ThreePhaseProcessorSimBlock

def performance_test():
    print("\n" + "="*70)
    print("PERFORMANCE COMPARISON")
    print("="*70)
    
    # Create source and input
    source = ThreePhaseGenerator("source", amplitude=10.0, freq=50.0)
    input_signal = source.compute(0, 0.001, None)
    
    # Create processors
    processor_c = ThreePhaseProcessorSimBlock("processor_c", use_c_backend=True)
    processor_py = ThreePhaseProcessorSimBlock("processor_py", use_c_backend=False)
    gain_py = VectorGain("gain_py", gain=2.0)  # Regular Python gain
    
    # Warm-up
    print("\nWarming up...")
    for _ in range(1000):
        processor_c.compute(0, 0.001, [input_signal])
        processor_py.compute(0, 0.001, [input_signal])
        gain_py.compute(0, 0.001, [input_signal])
    
    # Benchmark
    iterations = 100000
    print(f"\nBenchmarking {iterations} iterations...")
    
    # C version
    start = time.perf_counter()
    for _ in range(iterations):
        processor_c.compute(0, 0.001, [input_signal])
    c_time = time.perf_counter() - start
    
    # Python generated version
    start = time.perf_counter()
    for _ in range(iterations):
        processor_py.compute(0, 0.001, [input_signal])
    py_gen_time = time.perf_counter() - start
    
    # Regular Python gain
    start = time.perf_counter()
    for _ in range(iterations):
        gain_py.compute(0, 0.001, [input_signal])
    py_reg_time = time.perf_counter() - start
    
    # Results
    print("\n" + "="*70)
    print("RESULTS")
    print("="*70)
    print(f"C Version:           {c_time*1000:.3f} ms")
    print(f"Python Generated:    {py_gen_time*1000:.3f} ms")
    print(f"Python Regular:      {py_reg_time*1000:.3f} ms")
    print(f"\nSpeedup (C vs Python Generated): {py_gen_time/c_time:.1f}x")
    print(f"Speedup (C vs Python Regular):   {py_reg_time/c_time:.1f}x")
    
    # Plot
    plt.figure(figsize=(10, 6))
    versions = ['C', 'Python Gen', 'Python Reg']
    times = [c_time*1000, py_gen_time*1000, py_reg_time*1000]
    colors = ['#2E75B5', '#ED7D31', '#70AD47']
    
    bars = plt.bar(versions, times, color=colors)
    plt.ylabel('Time (ms)')
    plt.title(f'Performance Comparison ({iterations} iterations)')
    
    # Add value labels
    for bar, time in zip(bars, times):
        height = bar.get_height()
        plt.text(bar.get_x() + bar.get_width()/2., height,
                f'{time:.1f} ms', ha='center', va='bottom')
    
    plt.grid(True, alpha=0.3)
    plt.savefig('performance_comparison.png')
    plt.show()
    
    return c_time, py_gen_time, py_reg_time

if __name__ == "__main__":
    performance_test()

#!/usr/bin/env python3
"""
Deploy generated code to embedded targets
"""

from embedsim.code_generator import MCUTarget

def deploy_to_mcu():
    print("\n" + "="*70)
    print("DEPLOYING TO EMBEDDED TARGETS")
    print("="*70)
    
    print("\nAvailable MCU Targets:")
    print(f"  - AURIX Tricore: {MCUTarget.AURIX_TRICORE}")
    print(f"  - Cortex-M4:     {MCUTarget.CORTEX_M4}")
    
    print("\n" + "="*70)
    print("DEPLOYMENT STEPS")
    print("="*70)
    print("""
1. Copy generated files to your embedded project:
   - three_phase_processor.h
   - three_phase_processor.c

2. For AURIX Tricore:
   - Use TriCore compiler (tricore-gcc)
   - Add to Makefile:
     CC = tricore-gcc
     CFLAGS = -O2 -mcpu=tc1.6.2
     
   Compile:
     tricore-gcc -O2 -c three_phase_processor.c

3. For Cortex-M4:
   - Use arm-none-eabi-gcc
   - Add to Makefile:
     CC = arm-none-eabi-gcc
     CFLAGS = -O2 -mcpu=cortex-m4 -mfloat-abi=hard -mfpu=fpv4-sp-d16
     
   Compile:
     arm-none-eabi-gcc -O2 -c three_phase_processor.c

4. Link with your application:
   - Call three_phase_processor_compute() in your control loop
   - Pass input/output structs
    """)
    
    print("\nExample C caller code:")
    print("""
#include "three_phase_processor.h"

// In your control loop (e.g., 10kHz)
void control_loop(void) {
    static InputSignals in;
    static OutputSignals out;
    
    // Read ADC for three-phase currents
    in.source[0] = read_adc(0);  // Phase U
    in.source[1] = read_adc(1);  // Phase V  
    in.source[2] = read_adc(2);  // Phase W
    
    // Call generated function
    three_phase_processor_compute(&in, &out);
    
    // Use outputs
    set_dac(0, out.gain[0]);  // Phase U amplified
    set_dac(1, out.gain[1]);  // Phase V amplified
    set_dac(2, out.gain[2]);  // Phase W amplified
}
    """)

if __name__ == "__main__":
    deploy_to_mcu()

11. Troubleshooting

11.1 Common Issues and Solutions

Error Solution
ImportError: No module named 'embedsim' Make sure EmbedSim is in your Python path. Run:
set PYTHONPATH=C:\EmbedSimProject (Windows)
export PYTHONPATH=/path/to/EmbedSimProject (Linux)
ValueError: Algebraic loop detected Add VectorDelay in the feedback path. Import from simulation_engine:
from embedsim.simulation_engine import VectorDelay
error: Unable to find vcvarsall.bat Install Visual Studio Build Tools or use MinGW:
pip install --global-option build_ext --global-option --compiler=mingw32 cython
gcc: command not found Install gcc:
Windows: MinGW-w64
Linux: sudo apt install gcc
Mac: xcode-select --install
undefined symbol: three_phase_processor_compute Make sure you implemented the function in .c file and it's included in sources list
